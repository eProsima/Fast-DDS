    set(TESTS

    foreach(TEST IN LISTS TESTS)

                ${CMAKE_INSTALL_PREFIX}

            TEST system.tools.fastdds.${TEST}
            PROPERTY LABELS "NoMemoryCheck"
        )

    endforeach()
